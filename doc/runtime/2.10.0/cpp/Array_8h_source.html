479  const detail::ElementFactory<ARRAY_TRAITS>& elementFactory,
480  const OFFSET_CHECKER& offsetChecker)
481  {

483  "Implicit array elements must have constant bit size!");

486  if (ARRAY_TYPE == ArrayType::IMPLICIT)
