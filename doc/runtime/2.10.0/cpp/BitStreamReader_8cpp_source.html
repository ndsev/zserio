185  inline void loadCacheNext(ReaderContext& ctx, uint8_t numBits)
186  {
187  static const uint8_t cacheBitSize = sizeof(BaseType) * 8;
188  BaseType& cacheBuffer = getCacheBuffer(ctx.cache);
189 
190  // ctx.bitIndex is always byte aligned and ctx.cacheNumBits is always zero in this call
191  const size_t byteIndex = ctx.bitIndex >> 3;
192  if (ctx.bufferBitSize >= ctx.bitIndex + cacheBitSize)
193  {
194  cacheBuffer =
195 #ifdef ZSERIO_RUNTIME_64BIT
196  parse64(ctx.buffer + byteIndex);
197 #else
198  parse32(ctx.buffer + byteIndex);
199 #endif
200  ctx.cacheNumBits = cacheBitSize;
201  }
202  else
203  {
204  if (ctx.bitIndex + numBits > ctx.bufferBitSize)
205  throwEof();
206 
207  ctx.cacheNumBits = static_cast<uint8_t>(ctx.bufferBitSize - ctx.bitIndex);
208 
209  // buffer must be always available in full bytes, even if some last bits are not used
210  const uint8_t alignedNumBits = (ctx.cacheNumBits + 7) & ~0x7;
211 
212  switch (alignedNumBits)
213  {
214 #ifdef ZSERIO_RUNTIME_64BIT
215  case 64:
216  cacheBuffer = parse64(ctx.buffer + byteIndex);
217  break;
218  case 56:
219  cacheBuffer = parse56(ctx.buffer + byteIndex);
220  break;
221  case 48:
222  cacheBuffer = parse48(ctx.buffer + byteIndex);
223  break;
224  case 40:
225  cacheBuffer = parse40(ctx.buffer + byteIndex);
226  break;
227 #endif
228  case 32:
229  cacheBuffer = parse32(ctx.buffer + byteIndex);
230  break;
231  case 24:
232  cacheBuffer = parse24(ctx.buffer + byteIndex);
233  break;
234  case 16:
235  cacheBuffer = parse16(ctx.buffer + byteIndex);
236  break;

247  inline BaseType readBitsImpl(ReaderContext& ctx, uint8_t numBits)
248  {
249  BaseType value = 0;
250  const BaseType& cacheBuffer = getCacheBuffer(ctx.cache);
251 
252  if (ctx.cacheNumBits < numBits)
253  {
254  // read all remaining cache bits
255  value = cacheBuffer & MASK_TABLE[ctx.cacheNumBits];
256  ctx.bitIndex += ctx.cacheNumBits;
257  numBits -= ctx.cacheNumBits;
258 
259  // load next piece of buffer into cache
260  loadCacheNext(ctx, numBits);
261 
262  // add the remaining bits to the result
263  // if numBits is sizeof(BaseType) * 8 here, value is already 0 (see MASK_TABLE[0])
264  if (numBits < sizeof(BaseType) * 8)
265  value <<= numBits;
266  }
267  value |= ((cacheBuffer >> (ctx.cacheNumBits - numBits)) & MASK_TABLE[numBits]);
268  ctx.cacheNumBits -= numBits;
269  ctx.bitIndex += numBits;
270 
271  return value;
272  }
