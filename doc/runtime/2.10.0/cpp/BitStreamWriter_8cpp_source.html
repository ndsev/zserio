312 void BitStreamWriter::writeUnsignedBits(uint32_t data, uint8_t numBits)
313 {
314  if (!hasWriteBuffer())
315  {
316  m_bitIndex += numBits;
317  return;
318  }
319 
320  checkCapacity(m_bitIndex + numBits);
321 
322  uint8_t restNumBits = numBits;
323  const uint8_t bitsUsed = m_bitIndex & 0x07;
324  uint8_t bitsFree = 8 - bitsUsed;
325  size_t byteIndex = m_bitIndex / 8;
326 
327  if (restNumBits > bitsFree)
328  {
329  // first part
330  const uint8_t shiftNum = restNumBits - bitsFree;
331  const uint8_t maskedByte = m_buffer[byteIndex] & ~(0xFF >> bitsUsed);
332  m_buffer[byteIndex++] = maskedByte | static_cast<uint8_t>(data >> shiftNum);
333  restNumBits -= bitsFree;
334 
335  // middle parts
336  while (restNumBits >= 8)
337  {
338  restNumBits -= 8;
339  m_buffer[byteIndex++] = static_cast<uint8_t>((data >> restNumBits) & MAX_U32_VALUES[8]);
340  }
341 
342  // reset bits free
343  bitsFree = 8;
344  }
345 
346  // last part
347  if (restNumBits > 0)
348  {
349  const uint8_t shiftNum = bitsFree - restNumBits;
350  const uint32_t mask = MAX_U32_VALUES[restNumBits];
351  const uint8_t maskedByte = m_buffer[byteIndex] & ~static_cast<uint8_t>(mask << shiftNum);
352  m_buffer[byteIndex] = maskedByte | static_cast<uint8_t>((data & mask) << shiftNum);
353  }
354 
355  m_bitIndex += numBits;
356 }

382 inline void BitStreamWriter::writeVarNum(uint64_t value, bool isSigned, bool isNegative, size_t maxVarBytes,
383  size_t numVarBytes)
384 {
385  static const uint64_t bitMasks[8] = { 0x01, 0x03, 0x07, 0x0F, 0x1F, 0x3F, 0x7F, 0xFF };
386  const bool hasMaxByteRange = (numVarBytes == maxVarBytes);
387 
388  for (size_t i = 0; i < numVarBytes; i++)
389  {
390  uint8_t byte = 0x00;
391  uint8_t numBits = 8;
392  const bool hasNextByte = (i < numVarBytes - 1);
393  const bool hasSignBit = (isSigned && i == 0);
394  if (hasSignBit)
395  {
396  if (isNegative)
397  byte |= 0x80;
398  numBits--;
399  }
400  if (hasNextByte)
401  {
402  numBits--;
403  byte |= (0x01 << numBits); // use bit 6 if signed bit is present, use bit 7 otherwise
404  }
405  else // this is the last byte
406  {
407  if (!hasMaxByteRange) // next byte indicator is not used in last byte in case of max byte range
408  numBits--;
409  }
410 
411  const size_t shiftBits = (numVarBytes - (i + 1)) * 7 + ((hasMaxByteRange && hasNextByte) ? 1 : 0);
412  byte |= static_cast<uint8_t>((value >> shiftBits) & bitMasks[numBits - 1]);
413  writeUnsignedBits(byte, 8);
414  }
415 }
