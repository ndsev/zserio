17 // calculates bit length on delta provided as an absolute number
18 inline uint8_t absDeltaBitLength(uint64_t absDelta)
19 {
20  uint8_t result = 0;
21  while (absDelta > 0)
22  {

30 // calculates bit length, emulates Python bit_length to keep same logic
31 template <typename T>
32 uint8_t calcBitLength(T lhs, T rhs)
33 {
34  const uint64_t absDelta = lhs > rhs
35  ? static_cast<uint64_t>(lhs) - static_cast<uint64_t>(rhs)
36  : static_cast<uint64_t>(rhs) - static_cast<uint64_t>(lhs);
37 
38  return absDeltaBitLength(absDelta);
39 }

41 // calculates delta, doesn't check for possible int64_t overflow since it's used only in cases where it's

43 template <typename T>
44 int64_t calcUncheckedDelta(T lhs, uint64_t rhs)
45 {
46  return static_cast<int64_t>(static_cast<uint64_t>(lhs) - rhs);
