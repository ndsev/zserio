template<typename T, typename ELEMENT_BIT_SIZE>
class zserio::DynamicBitFieldArrayTraits< T, ELEMENT_BIT_SIZE >

Array traits for dynamic bit field Zserio types with dynamic bit size (int<bitSize> or bit<bitSize>).
