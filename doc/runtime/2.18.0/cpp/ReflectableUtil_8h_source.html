300 inline bool ReflectableUtil::doubleValuesAlmostEqual(double lhs, double rhs)
301 {
302  if (std::isinf(lhs) || std::isinf(rhs))
303  {
304  return std::isinf(lhs) && std::isinf(rhs) && ((lhs > 0.0 && rhs > 0.0) || (lhs < 0.0 && rhs < 0.0));
305  }
306 
307  if (std::isnan(lhs) || std::isnan(rhs))
308  {
309  return std::isnan(lhs) && std::isnan(rhs);
310  }
311 
312  // see: https://en.cppreference.com/w/cpp/types/numeric_limits/epsilon
313  return std::fabs(lhs - rhs) <= std::numeric_limits<double>::epsilon() * std::fabs(lhs + rhs) ||
314  std::fabs(lhs - rhs) < std::numeric_limits<double>::min();
